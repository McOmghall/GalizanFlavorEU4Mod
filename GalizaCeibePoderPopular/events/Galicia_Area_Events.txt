namespace = galician_area

# The irmandiño wars, a peasant revolt against privileges of nobility: Starting event.
country_event = {
	id = galician_area.1
	title = "galician_area.EVTNAME1"
	desc = "galician_area.EVTDESC1"
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
    any_owned_province = {
      area = galicia_area
    }
    is_year = 1444
		NOT = { is_year = 1500 }
		has_estate = estate_nobles
	}

	mean_time_to_happen = {
		# The great brotherhood rises in 1467
		# Usually Castille will get modifiers by 0.8 * 0.8 * 0.5
		months = 870

		# Unstable countries get revolts faster
		modifier = {
			factor = 0.8
			NOT = { stability = 0 }
		}

		# A powerful local nobility stirs peasants to revolt
		# Default Castille gets this
		modifier = {
			factor = 0.8
			estate_influence = {
    		estate = estate_nobles
    		influence = 50
			}
		}

		# A more powerful local nobility stirs more peasants to revolt
		# Default Castille gets this
		modifier = {
			factor = 0.8
			estate_influence = {
    		estate = estate_nobles
    		influence = 70
			}
		}

		# England helped the irmandiños to revolt
		# Usually Default Castille gets this
		modifier = {
			factor = 0.5
			OR = {
				ENG = {
					NOT = {
						has_opinion = {
							who = ROOT
							value = 80
						}
					}
				}
				AND = {
					GBR = {
						NOT = {
							has_opinion = {
								who = ROOT
								value = 80
							}
						}
					}
					NOT = { exists = ENG }
				}
			}
		}

		# England helped the irmandiños to revolt, if they're your friend then it's no problem
		modifier = {
			factor = 100
			OR = {
				ENG = {
					has_opinion = {
						who = ROOT
						value = 80
					}
				}
				AND = {
					GBR = {
						has_opinion = {
							who = ROOT
							value = 80
						}
					}
					NOT = { exists = ENG }
				}
			}
		}
	}

	option = {
		# Help local nobility against the irmandiños
		name = "galician_area.EVTOPTA1"
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 30
		}
		add_prestige = 5
		set_country_flag = irmandinho_war_started
		random_owned_province = {
			limit = {
				area = galicia_area
			}
			spawn_rebels = {
				friend = GAL
				type = irmandinhos
				size = 2
				win = yes
				leader = "Alfonso de Lanzós"
				culture = THIS
			}
		}
	}

	option = {
		# Seize the opportunity to curtail noble perrogative and reinforce the authority of the king
		name = "galician_area.EVTOPTB1"
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -40
		}
		add_legitimacy = 20
		add_prestige = -5
		add_country_modifier = {
			name = irmandinhos_used_against_nobility
			duration = 3650
			desc = irmandinhos_used_against_nobility_tooltip
			hidden = no
		}
		set_country_flag = irmandinho_war_started
		random_owned_province = {
			limit = {
				area = galicia_area
			}
			spawn_rebels = {
				friend = GAL
				type = irmandinhos
				size = 3
				win = yes
				leader = "Alfonso de Lanzós"
				culture = THIS
			}
		}
	}

	option = {
		# Negotiate
		name = "galician_area.EVTOPTC1"
		add_prestige = -10
		add_legitimacy = 10
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_estate_influence = {
			estate = estate_nobles
			influence = -10
		}
		random_owned_province = {
			limit = {
				area = galicia_area
			}
			add_province_modifier = {
				name = irmandinhos_negotiated_with
				duration = 3650
				desc = irmandinhos_negotiated_with_tooltip
				hidden = no
			}
		}
	}
}

# The irmandiño wars, a peasant revolt against privileges of nobility: Ending event.
country_event = {
	id = galician_area.100
	title = "galician_area.EVTNAME100"
	desc = "galician_area.EVTDESC100"
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
    has_country_flag = irmandinho_war_started
    NOT = { has_country_flag = irmandinho_war_lost }
    all_owned_province = {
			controlled_by = ROOT
      NOT = { has_province_modifier = irmandinhos_embolden_peasantry }
    }
		any_owned_province = {
			has_province_flag = was_occupied_by_irmandinhos
		}
	}

	mean_time_to_happen = {
		days = 10
	}

	option = { # Winning the irmandinho wars paves the way for greater centralization and reduced local noble powers in the occupied provinces
		name = "galician_area.EVTOPTA100"

		add_legitimacy = 10
    every_owned_province = {
      limit = {
        has_province_flag = was_occupied_by_irmandinhos
      }
      add_province_modifier = {
        name = irmandinhos_weakened_local_nobility
        duration = -1
				desc = irmandinhos_weakened_local_nobility_tooltip
				hidden = no
      }
    }
	}
}
